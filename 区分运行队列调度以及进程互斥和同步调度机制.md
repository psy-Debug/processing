1.1 信号量的实现（简化）
c

struct semaphore {
    int count;                    // 资源数量
    struct list_head wait_list;   // 等待队列
};

// 申请资源（可能阻塞）
void down(struct semaphore *sem) {
    if (sem->count > 0) {
        sem->count--;              // 有资源，直接拿走
        return;
    }
    // 没资源了，当前进程必须阻塞
    set_current_state(TASK_INTERRUPTIBLE);  // 设置阻塞状态
    list_add(&current->task_list, &sem->wait_list);  // 加入等待队列
    schedule();  // ← 这里调用总调度器，切换到另一个进程
}

// 释放资源（可能唤醒）
void up(struct semaphore *sem) {
    if (list_empty(&sem->wait_list)) {
        sem->count++;               // 没人等，增加资源计数
        return;
    }
    // 有人等在等，唤醒第一个
    task = list_first_entry(&sem->wait_list);
    list_del(&task->task_list);
    wake_up_process(task);  // ← 这里调用总调度器，把任务放进就绪队列
}

// 总调度器不知道也不关心进程在等什么资源
// 它只关心：谁在就绪队列里，该让谁运行

// 同步机制负责：
// 1. 资源不够时，把进程移出就绪队列（阻塞）

// 当进程被移出/放回就绪队列时，需要调用总调度器
// 因为就绪队列变了，可能需要重新选择运行的进程

三、阻塞和激活的完整流程
3.1 场景：两个进程访问同一把锁
c

// 进程A先拿到锁
mutex_lock(&lock);    // count从1变0，成功返回
// 进程A正在临界区

// 进程B也想拿锁
mutex_lock(&lock);    // count已经是0，进入阻塞流程

// 阻塞流程：
mutex_lock() {
    // 1. 发现锁被占用
    // 2. 把进程B加入等待队列
    // 3. 设置进程B为 TASK_INTERRUPTIBLE（阻塞状态）
    // 4. 调用 schedule()  ← 总调度器被调用
}

// 总调度器执行：
schedule() {
    // 进程B被移出就绪队列，进程A还在就绪队列
    // 选择下一个进程 → 选到进程A
    // 切换回进程A继续运行
}

// 进程A释放锁
mutex_unlock(&lock) {
    // 1. 发现有人在等
    // 2. 把进程B从等待队列移除
    // 3. 设置进程B为 TASK_RUNNING（就绪状态）
    // 4. 把进程B放回就绪队列
    // 5. 可能调用 schedule() 让高优先级进程抢占
}

4.2 你只需要知道

    就绪队列：总调度器管理的，存放可运行的进程

    等待队列：同步机制管理的，存放等待资源的进程

    阻塞：进程从就绪队列移到等待队列

    唤醒：进程从等待队列移到就绪队列

    切换：总调度器在就绪队列里选下一个

// 4. 进程主动让出
sched_yield() → schedule()
